local HttpService = game:GetService("HttpService")

local xor = function(str, key)
    local result = {}
    for i = 1, #str do
        local keyChar = key:byte((i - 1) % #key + 1)
        result[i] = string.char(bit32.bxor(str:byte(i), keyChar))
    end
    return table.concat(result)
end
